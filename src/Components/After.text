import React, { useState, useMemo, useRef, useEffect } from 'react';
import axios from 'axios';
import {
  IconRefresh,
  IconUpload,
  IconDownload,
  IconCalendarEvent,
  IconUsers,
  IconWorld,
  IconPlanet,
  IconBuildingBank,
  IconDiamond,
  IconCrown,
  IconSearchOff,
  IconEdit,
  IconTrash,
  IconX
} from '@tabler/icons-react';
import { Chart } from 'chart.js/auto';
import UploadModal from "./UploadModal";
import DeleteConfirm from "./DeleteConfirm";

// --- Investor Meta ---
const IM = [
  { k: "Indian Resident", cls: "bg-gray-50 border-gray-200 text-gray-600", lbl: "Indian Res." },
  { k: "US NRI", cls: "bg-indigo-50 border-indigo-200 text-indigo-600", lbl: "US NRI" },
  { k: "Canadian NRI", cls: "bg-red-50 border-red-200 text-red-600", lbl: "CA NRI" },
  { k: "UK NRI", cls: "bg-blue-50 border-blue-200 text-blue-600", lbl: "UK NRI" },
  { k: "Singapore NRI", cls: "bg-teal-50 border-teal-200 text-teal-600", lbl: "SG NRI" },
  { k: "UAE NRI", cls: "bg-emerald-50 border-emerald-200 text-emerald-600", lbl: "UAE NRI" },
  { k: "Foreign Investor", cls: "bg-orange-50 border-orange-200 text-orange-600", lbl: "Foreign" },
  { k: "Institution", cls: "bg-cyan-50 border-cyan-200 text-cyan-600", lbl: "Institution" },
  { k: "HNI", cls: "bg-amber-50 border-amber-200 text-amber-600", lbl: "HNI" },
  { k: "UHNI", cls: "bg-purple-50 border-purple-200 text-purple-600", lbl: "UHNI / FO" },
];
const ICM = Object.fromEntries(IM.map(x => [x.k, x]));

const ProductManager = () => {
  const [P, setP] = useState([]);
  const [monthlyData, setMonthlyData] = useState([]);
  const [loading, setLoading] = useState(true);
  //Upload pdf
  const [isUploadModalOpen, setUploadModalOpen] = useState(false);

  // Delete confirm
  const [deleteConfirmOpen, setDeleteConfirmOpen] = useState(false);
  const [productToDelete, setProductToDelete] = useState(null);

  // Drawer State
  const [selectedProduct, setSelectedProduct] = useState(null);
  const [selectedMonthlyData, setSelectedMonthlyData] = useState([]);

  useEffect(() => {
    if (selectedProduct) {
      axios.get(`http://localhost:5000/api/monthly-data/${selectedProduct.id}`)
        .then(res => {
          setSelectedMonthlyData(res.data);
        })
        .catch(err => console.error("Failed to fetch monthly data for product", err));
    } else {
      setSelectedMonthlyData([]);
    }
  }, [selectedProduct]);

  // --- Filters State ---
  const [activeTab, setActiveTab] = useState('list');
  const [investorPills, setInvestorPills] = useState(['all']);

  // Dropdowns
  const [search, setSearch] = useState('');
  const [filterType, setFilterType] = useState('');
  const [filterCategory, setFilterCategory] = useState('');
  const [filterApproach, setFilterApproach] = useState('');
  const [filterCapBias, setFilterCapBias] = useState('');
  const [filterCurrency, setFilterCurrency] = useState('');
  const [filterRevModel, setFilterRevModel] = useState('');
  const [filterAMC, setFilterAMC] = useState('');

  // Sliders
  const [fSh, setFSh] = useState(0);
  const [fSd, setFSd] = useState(40);
  const [fUc, setFUc] = useState(50);
  const [fDc, setFDc] = useState(120);
  const [fAm, setFAm] = useState(0);
  const [fR3, setFR3] = useState(-10);
  const [fTr, setFTr] = useState(0);
  const [fMi, setFMi] = useState('');

  // Compare
  const [compareIds, setCompareIds] = useState([]);

  // Sorting
  const [sortKey, setSortKey] = useState(null);
  const [sortDir, setSortDir] = useState(1);

  const chartRef = useRef(null);

  // --- Derived Data ---
  const amcOptions = useMemo(() => Array.from(new Set(P.map(p => p.amc))).sort(), [P]);

  const fetchDashboardData = async () => {
    try {
      setLoading(true);
      const res = await axios.get('http://localhost:5000/api/dashboard-products');
      setP(res.data);
      console.log(res.data);

      const mdRes = await axios.get('http://localhost:5000/api/monthly-data');
      setMonthlyData(mdRes.data);

    } catch (err) {
      console.error("Failed to load dashboard data", err);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchDashboardData();
  }, []);

  const handleReset = () => {
    setInvestorPills(['all']);
    setSearch('');
    setFilterType(''); setFilterCategory(''); setFilterApproach(''); setFilterCapBias('');
    setFilterCurrency(''); setFilterRevModel(''); setFilterAMC('');
    setFSh(0); setFSd(40); setFUc(50); setFDc(120); setFAm(0); setFR3(-10); setFTr(0); setFMi('');
    setCompareIds([]);
    setSortKey(null);
  };

  const handlePillClick = (k) => {
    if (k === 'all') {
      setInvestorPills(['all']);
      return;
    }
    let newPills = investorPills.filter(p => p !== 'all');
    if (newPills.includes(k)) {
      newPills = newPills.filter(p => p !== k);
      if (newPills.length === 0) newPills = ['all'];
    } else {
      newPills.push(k);
    }
    setInvestorPills(newPills);
  };

  //UPLOAD
  const handlePdf = () => {
    setUploadModalOpen(true);
  };

  const handleUploadSuccess = (data) => {
    console.log("Upload Success:", data);
    alert(`Successfully processed! Batch ID: ${data.batchId}`);
    fetchDashboardData();
  };

  const handleDeleteProduct = (productId, productName) => {
    setProductToDelete({ id: productId, name: productName });
    setDeleteConfirmOpen(true);
  };

  const executeDelete = async () => {
    if (!productToDelete) return;
    try {
      await axios.delete(`http://localhost:5000/api/product/${productToDelete.id}`);
      setDeleteConfirmOpen(false);
      setProductToDelete(null);
      fetchDashboardData();
    } catch (err) {
      console.error('Delete failed:', err);
      alert('Failed to delete product. ' + (err.response?.data?.error || err.message));
    }
  };

  const filteredProducts = useMemo(() => {
    return P.filter(p => {
      // Pills
      if (!investorPills.includes('all')) {
        const match = investorPills.some(inv => (p.elig || []).includes(inv));
        if (!match) return false;
      }
      // Dropdowns
      if (filterType && p.type !== filterType) return false;
      if (filterCategory && p.category !== filterCategory) return false;
      if (filterApproach && p.approach !== filterApproach) return false;
      if (filterCapBias && p.capBias !== filterCapBias) return false;
      if (filterCurrency && p.currency !== filterCurrency) return false;
      if (filterRevModel && p.revModel !== filterRevModel) return false;
      if (filterAMC && p.amc !== filterAMC) return false;

      // Sliders
      if ((p.sharpe || 0) < fSh) return false;
      if (p.stddev !== null && p.stddev !== undefined && p.stddev !== '' && parseFloat(p.stddev) > fSd) return false;
      if (p.upCap && p.upCap < fUc) return false;
      if (p.downCap && p.downCap > fDc) return false;
      if ((p.aum || 0) < fAm) return false;
      if (p.ret3 && p.ret3 < fR3) return false;
      if ((p.trail || 0) < fTr) return false;

      // Min Inv
      if (fMi) {
        const limit = parseFloat(fMi);
        if ((p.minInvL || 9999) > limit) return false;
      }

      // Search
      if (search) {
        const s = search.toLowerCase();
        if (!p.name?.toLowerCase().includes(s) &&
          !p.amc?.toLowerCase().includes(s) &&
          !p.manager?.toLowerCase().includes(s)) {
          return false;
        }
      }

      return true;
    }).sort((a, b) => {
      if (!sortKey) return 0;
      let valA = a[sortKey];
      let valB = b[sortKey];
      if (valA === null || valA === undefined) valA = -99999;
      if (valB === null || valB === undefined) valB = -99999;

      if (typeof valA === 'string') return sortDir * valA.localeCompare(valB);
      return sortDir * (valA - valB);
    });
  }, [P, investorPills, filterType, filterCategory, filterApproach, filterCapBias, filterCurrency, filterRevModel, filterAMC, fSh, fSd, fUc, fDc, fAm, fR3, fTr, fMi, search, sortKey, sortDir]);

  // Handle Sort
  const doSort = (k) => {
    if (sortKey === k) {
      setSortDir(d => d * -1);
    } else {
      setSortKey(k);
      setSortDir(1);
    }
  };

  // Stats computation
  const stats = useMemo(() => {
    const valid = filteredProducts;
    if (!valid.length) return { cnt: 0, sh: '—', r3: '—', dc: '—', au: '—', tr: '—', ex: '—' };

    let sumSh = 0, nSh = 0;
    let sumR3 = 0, nR3 = 0;
    let sumDc = 0, nDc = 0;
    let sumAu = 0;
    let sumTr = 0, nTr = 0;
    let sumEx = 0, nEx = 0;

    valid.forEach(p => {
      if (p.sharpe) { sumSh += parseFloat(p.sharpe); nSh++; }
      if (p.ret3) { sumR3 += parseFloat(p.ret3); nR3++; }
      if (p.downCap) { sumDc += parseFloat(p.downCap); nDc++; }
      if (p.aum) { sumAu += parseFloat(p.aum); }
      if (p.trail) { sumTr += parseFloat(p.trail); nTr++; }
      if (p.expRatio) { sumEx += parseFloat(p.expRatio); nEx++; }
    });

    return {
      cnt: valid.length,
      sh: nSh ? (sumSh / nSh).toFixed(2) : '—',
      r3: nR3 ? (sumR3 / nR3).toFixed(1) + '%' : '—',
      dc: nDc ? (sumDc / nDc).toFixed(0) + '%' : '—',
      au: sumAu ? '₹' + sumAu.toLocaleString() + 'Cr' : '—',
      tr: nTr ? (sumTr / nTr).toFixed(2) + '%' : '—',
      ex: nEx ? (sumEx / nEx).toFixed(2) + '%' : '—',
    };
  }, [filteredProducts]);

  // Initialize Risk Chart
  useEffect(() => {
    if (activeTab === 'risk') {
      const ctx = document.getElementById('rc');
      if (chartRef.current) chartRef.current.destroy();

      const dataPoints = filteredProducts.filter(p => p.stddev && p.sharpe).map(p => ({
        x: parseFloat(p.stddev),
        y: parseFloat(p.sharpe),
        r: Math.max(4, Math.min(25, (parseFloat(p.aum) || 100) / 200)),
        name: p.name,
        type: p.type || '—'
      }));

      // Group by type for coloring
      const dsMap = {};
      const colors = {
        "PMS": "rgba(201, 152, 26, 0.7)",
        "AIF Cat II": "rgba(124, 58, 237, 0.7)",
        "AIF Cat III": "rgba(26, 86, 219, 0.7)",
        "GIFT City – AIF": "rgba(14, 122, 69, 0.7)",
        "International FoF": "rgba(184, 92, 0, 0.7)",
        "Mutual Fund": "rgba(185, 28, 28, 0.7)"
      };

      dataPoints.forEach(pt => {
        if (!dsMap[pt.type]) dsMap[pt.type] = [];
        dsMap[pt.type].push(pt);
      });

      const datasets = Object.keys(dsMap).map(type => ({
        label: type,
        data: dsMap[type],
        backgroundColor: colors[type] || "rgba(150,150,150,0.7)"
      }));

      chartRef.current = new Chart(ctx, {
        type: 'bubble',
        data: { datasets },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            tooltip: {
              callbacks: {
                label: (ctx) => `${ctx.raw.name} | StdDev: ${ctx.raw.x}% | Sharpe: ${ctx.raw.y}`
              }
            }
          },
          scales: {
            x: { title: { display: true, text: 'Std Deviation (Risk %)' } },
            y: { title: { display: true, text: 'Sharpe Ratio (Return/Risk)' } }
          }
        }
      });
    }
  }, [activeTab, filteredProducts]);

  if (loading) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-[#F8F7F4]">
        <div className="animate-spin w-8 h-8 border-4 border-[#C9981A] border-t-transparent rounded-full"></div>
      </div>
    );
  }

  return (
    <div className="w-full min-h-screen bg-white text-[#1A1A1A] font-sans p-4 md:p-6 pb-20">
      <div className="max-w-[1500px] mx-auto">

        {/* HEADER */}
        <div className="flex flex-col md:flex-row justify-between items-start md:items-end mb-6 pb-4 border-b border-[#E0DDD6]">
          <div className="mb-4 md:mb-0">
            <div className="text-[10px] font-bold tracking-widest text-[#9E9A94] uppercase mb-1">TieVista · Product Universe</div>
            <div className="text-3xl font-light tracking-tight mb-1 text-gray-900">Investment Product Dashboard</div>
            <div className="text-sm text-[#6B6760]">PMS · AIF · GIFT City · International · Mutual Funds</div>
          </div>
          <div className="flex items-center gap-3">
            <button className="inline-flex items-center gap-1.5 px-3 py-1.5 text-sm font-medium rounded-lg border border-[#C8C4BC] bg-white hover:bg-[#F2F1ED] transition shadow-sm text-gray-700" onClick={handleReset}>
              <IconRefresh size={16} /> Reset all
            </button>
            <button
              onClick={handlePdf}
              className="inline-flex items-center gap-1.5 px-3 py-1.5 text-sm font-medium rounded-lg border border-[#C9981A] text-[#C9981A] hover:bg-yellow-50 transition shadow-sm">
              <IconUpload size={16} /> Upload PDF
            </button>

            <UploadModal
              isOpen={isUploadModalOpen}
              onClose={() => setUploadModalOpen(false)}
              onSuccess={handleUploadSuccess}
            />
            <DeleteConfirm
              isOpen={deleteConfirmOpen}
              onClose={() => { setDeleteConfirmOpen(false); setProductToDelete(null); }}
              onConfirm={executeDelete}
              productName={productToDelete?.name}
            />
            <button className="inline-flex items-center gap-1.5 px-3 py-1.5 text-sm font-medium rounded-lg bg-[#C9981A] text-white hover:bg-[#b38617] border border-[#b38617] transition shadow-sm">
              <IconDownload size={16} /> Export CSV
            </button>
          </div>
        </div>

        {/* FILTERS SECTION */}
        <div className="bg-white rounded-xl shadow-sm border border-[#E0DDD6] p-5 mb-6">
          <div className="flex items-center justify-between mb-4 pb-4 border-b border-[#E0DDD6]">
            <div className="flex items-center gap-2">
              <span className="text-[10px] font-bold tracking-[0.06em] text-[#9E9A94] uppercase">
                Data as of
              </span>
              <div className="inline-block px-2 py-0.5 bg-[#F2F1ED] text-[#6B6760] rounded text-[11px] font-bold">
                March 2024
              </div>
            </div>
            <div className="text-[11px] text-[#6B6760] flex items-center gap-1.5">
              <IconCalendarEvent size={14} />
              <span className="font-semibold text-[#C9981A]">Live Data</span>
              <span>from Database</span>
            </div>
          </div>

          <div className="text-xs font-bold text-[#6B6760] mb-3 flex items-center gap-1.5 uppercase tracking-wider">
            <IconUsers size={14} /> Investable by — select investor type to filter
          </div>
          <div className="flex flex-wrap gap-2 mb-6">
            <div className={`px-3 py-1.5 rounded-full border text-xs font-medium cursor-pointer transition flex items-center gap-1.5 ${investorPills.includes('all') ? 'bg-[#1A1A1A] border-[#1A1A1A] text-white hover:bg-black' : 'border-[#E0DDD6] text-[#6B6760] hover:border-[#C9981A] hover:bg-[#FDF8E7]'}`} onClick={() => handlePillClick('all')}>
              <IconWorld size={14} /> All investors
            </div>
            {IM.map(i => {
              const isOn = investorPills.includes(i.k);
              return (
                <div key={i.k} className={`px-3 py-1.5 rounded-full border text-xs font-medium cursor-pointer transition flex items-center gap-1.5 ${isOn ? 'bg-[#1A1A1A] border-[#1A1A1A] text-white' : 'border-[#E0DDD6] text-[#6B6760] hover:border-[#C9981A] hover:bg-[#FDF8E7]'}`} onClick={() => handlePillClick(i.k)}>
                  {i.k.includes('Foreign') ? <IconPlanet size={14} /> :
                    i.k.includes('Institution') ? <IconBuildingBank size={14} /> :
                      i.k.includes('HNI') ? <IconDiamond size={14} /> :
                        i.k.includes('UHNI') ? <IconCrown size={14} /> : ''}
                  {i.lbl}
                </div>
              );
            })}
          </div>

          {/* Dropdowns */}
          <div className="flex flex-wrap gap-4 mb-4">
            <div className="flex flex-col flex-1 min-w-[200px]" style={{ flex: 2 }}>
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Search</div>
              <input type="text" className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A] placeholder-gray-400" placeholder="Fund name, manager, strategy…" value={search} onChange={e => setSearch(e.target.value)} />
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Product type</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterType} onChange={e => setFilterType(e.target.value)}>
                <option value="">All types</option>
                <option>PMS</option>
                <option>AIF Cat I</option>
                <option>AIF Cat II</option>
                <option>AIF Cat III</option>
                <option>GIFT City – PMS</option>
                <option>GIFT City – AIF</option>
                <option>GIFT City – MF</option>
                <option>International MF</option>
                <option>International FoF</option>
                <option>Mutual Fund</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Strategy</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterCategory} onChange={e => setFilterCategory(e.target.value)}>
                <option value="">All</option>
                <option>Large Cap Equity</option>
                <option>Flexi Cap</option>
                <option>Multi Cap</option>
                <option>Mid & Small Cap</option>
                <option>Multi Asset</option>
                <option>Credit Strategy</option>
                <option>Absolute Return</option>
                <option>Long-Short Equity</option>
                <option>Global Equity</option>
                <option>International Debt</option>
                <option>Thematic / Sectoral</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Approach</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterApproach} onChange={e => setFilterApproach(e.target.value)}>
                <option value="">All</option>
                <option>Value</option>
                <option>Growth</option>
                <option>GARP</option>
                <option>Blend</option>
                <option>Top-Down</option>
                <option>Bottom-Up</option>
                <option>Quant</option>
                <option>Multi-Factor</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Cap bias</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterCapBias} onChange={e => setFilterCapBias(e.target.value)}>
                <option value="">Any</option>
                <option>Large Cap</option>
                <option>Mid Cap</option>
                <option>Small Cap</option>
                <option>Multi Cap</option>
                <option>N/A (Debt/Credit)</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Currency</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterCurrency} onChange={e => setFilterCurrency(e.target.value)}>
                <option value="">Any</option>
                <option>INR</option>
                <option>USD</option>
                <option>EUR</option>
                <option>GBP</option>
                <option>AED</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[140px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Rev model</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterRevModel} onChange={e => setFilterRevModel(e.target.value)}>
                <option value="">Any</option>
                <option>Trail Only</option>
                <option>Upfront Only</option>
                <option>Upfront + Trail</option>
                <option>Trail + Perf Share</option>
                <option>Upfront + Trail + Perf</option>
              </select>
            </div>
            <div className="flex flex-col flex-1 min-w-[160px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">AMC / Fund house</div>
              <select className="w-full text-sm p-2 rounded border border-[#E0DDD6] bg-white focus:outline-none focus:border-[#C9981A] focus:ring-1 focus:ring-[#C9981A]" value={filterAMC} onChange={e => setFilterAMC(e.target.value)}>
                <option value="">All AMCs</option>
                {amcOptions.map(amc => <option key={amc}>{amc}</option>)}
              </select>
            </div>
          </div>

          {/* Sliders */}
          <div className="flex flex-wrap gap-4 mt-6 pt-4 border-t border-dashed border-[#E0DDD6]">
            <div className="flex flex-col flex-1 min-w-[120px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Sharpe ≥</div>
              <div className="flex items-center gap-2">
                <input className="w-full accent-[#C9981A]" type="range" min="0" max="3" step="0.1" value={fSh} onChange={e => setFSh(parseFloat(e.target.value))} />
                <span className="text-xs font-medium w-8 text-right text-gray-700">{fSh.toFixed(1)}</span>
              </div>
            </div>
            <div className="flex flex-col flex-1 min-w-[120px]">
              <div className="text-[11px] font-bold text-[#9E9A94] uppercase tracking-wider mb-1.5">Std dev ≤ (%)</div>
              <div className="flex items-center gap-2">
                <input className="w-full accent-[#C9981A]" type="range" min="5" max="40" step="1" value={fSd} onChange={e => setFSd(parseFloat(e.target.value))} />
                <span className="text-xs font-medium w-8 text-right text-gray-700">{fSd}%</span>
              </div>
            </div>
